class Item { 
	name;
	leafs = [];
    
	constructor(name, leafs) {					
		this.name = name;
		this.leafs = leafs;
	}
    
    getName() {
    	return this.name;
    }
    
    getLeafs() {
    	return this.leafs;
    }
}
